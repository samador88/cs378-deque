45 template <typename A, typename II, typename BI>
46 BI uninitialized_copy (A& a, II b, II e, BI x) {
47  BI p = x;
48  try {
49  while (b != e) {
50  a.construct(&*x, *b);
51  ++b;
52  ++x;}}
53  catch (...) {
54  destroy(a, p, x);
55  throw;}
56  return x;}

549  explicit my_deque (size_type s, const_reference v = value_type(), const allocator_type& a = allocator_type())
550  : _a(a), _aP() {
551  assert (s>=0);
552  size_type numrows = s / 10; //divide the size into different rows
553  assert(numrows >= 0);
554  if (s % 10 > 0){ //make sure there is no extra space, add an extra row if remainder
555  ++numrows;
556 
557  _outermost = _aP.allocate(numrows);
558 
559  size_type x = 0;
560  while (x != numrows){
561  _outermost[x] = _a.allocate(10);
562  ++x;
563  }
564 
565  x = 0;
566  while(x != numrows){
567  _a.deallocate(_outermost[x], 10);
568  ++x;
569  }
570 
571  _aP.deallocate(_outermost, numrows);
572  _bd = _lb = _a.allocate(s);
573 
574  _ed = _le = _bd + s;
575 
576  uninitialized_fill(_a, begin(), end(), v);
577  assert(valid());
578  }
579 
580  else{
581  _outermost = _aP.allocate(numrows);
582 
583  size_type x = 0;

895  void resize (size_type s, const_reference v = value_type()) {
896  if (s == size()){
897  assert(s==size());
898  return;
899  }
900  if (s < size()){
901  assert(s<size());
902  _ed = &(*destroy(_a, begin() + s, end() )); //had to put &* to compile
903  }
904  else if (s <= (_ed -_lb)){
905  assert(s<=(_ed -_lb));
906  _ed = &*uninitialized_fill(_a, end(), begin() + s, v);
907  }
908  else { //need to make new, bigger deque but keep all pointers the same
909  assert(!(s<=(_ed -_lb)));
910  assert(!(s<size()));
911  assert(!(s==size()));
912  size_type newsize = size() * 2;
913  if (s > newsize){
914  assert(s>newsize);
915  newsize = s;
916  }
917 
918  if (newsize == s){
919  assert(s==newsize);
920  newsize = newsize + 10;
921  }
922 
923  size_type middled = (newsize - s) / 2;
924  size_type enddata = middled;
925 
926  if ((newsize - s) % 2){
927  assert((newsize-s)%2);
928  ++enddata;
929  }
930 
931  my_deque x(newsize, v);
932 
933  std::copy(begin(), end(), x.begin() + middled);
934 
935  destroy(x._a, x.begin(), x.begin() + middled);
936  destroy(x._a, x.end() - enddata, x.end());
937  x._bd = x._bd + middled;
938  x._ed = x._ed - enddata;

940  swap(x);}
